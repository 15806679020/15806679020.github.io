  • 问题描述：

    • 文件上传漏洞通常由于代码中对文件上传功能所上传的文件过滤不严或web服务器相关解析漏洞未修复而造成的，如果文件上传功能代码没有严格限制和验证用户上传的文件后缀、类型等，攻击者可通过文件上传点上传任意文件，包括网站后门文件（webshell）控制整个网站。

  • 安全建议

    • 对上传文件类型进行验证，除在前端验证外在后端依然要做验证，后端可以进行扩展名检测，重命名文件，MIME类型检测以及限制上传文件的大小等限制来防御，或是将上传的文件其他文件存储服务器中。
    • 对上传文件内容进行过滤和校验。

     * @param inputStream
     * @return
     * @throws IOException

    public static boolean isFile(InputStream inputStream) throws IOException {
        byte[] byteArray = IOUtils.toByteArray(inputStream);
        String str = new String(byteArray);
        // 匹配16进制中的 <% ( ) %>
        // 匹配16进制中的 <? ( ) ?>
        // 匹配16进制中的 <script | /script> 大小写亦可
        // 通过匹配十六进制代码检测是否存在木马脚本
        String pattern = "(<%.*%>)|(<\\?.*\\?>)|(<script.*/script>)";
        Pattern mPattern = Pattern.compile(pattern);
        Matcher mMatcher = mPattern.matcher(str);

        // 查找相应的字符串
        boolean flag = true;
        if(mMatcher.find()) {
            //过滤java关键字(java import String print write( read() php request alert system)
            String keywordPattern = "(@eval\\()|(java)|(import)|(String)|(print)|(write)|(read\\(\\))|(php)|(request)|(alert)|(system)";
            Pattern keywordmPattern = Pattern.compile(keywordPattern);
            String keywordStr = mMatcher.group();
            Matcher keywordmMatcher = keywordmPattern.matcher(keywordStr);
            if(keywordmMatcher.find()){
                flag = false;
            }
        }
        return flag;
    }

     * 后缀判断
     * @param fileType    文件类型
     * @return
     * @throws Exception

    public static int suffixJudgement(String fileType) throws Exception {
        //全部转换为小写
        fileType = fileType.toLowerCase();
        //1.图片  2.文档 3.视频 5.音频 6.其他
        int type = 0;

        //其他
        String other = "txt,html,zip,cvs,rtf";

        //常见文档类型
        String file = "doc,docx,ppt,pptx,xls,xlsx,pdf";

        //常见视频格式
        String video = "mp4,flv,avi,mpg,wmv,mov,asx,asf,3gp";

        //图片
        String photo = "bmp,jpg,wbmp,jpeg,png,gif,psd,ico";

        //音频
        String audio = "mp3,wav";

        if(photo.contains(fileType))
            type = 1;

        if(file.contains(fileType))
            type = 2;

        if(video.contains(fileType))
            type = 3;

        if(audio.contains(fileType))
            type = 5;

        if(other.contains(fileType))
            type = 6;

        return type;
    }

        /** 
     * 判断文件类型 (并返回文件的类型编号)
     * @param path   文件路径
     * @return 文件类型 
     */  
    public static Integer getTypeNum(String path) throws IOException,Exception {  

        FileInputStream is = new FileInputStream(path);

        Integer typeNum = 0;
        String fileHead = getFileContent(is);  

        if (fileHead == null || fileHead.length() == 0) {  
            return null;  
        }  

        fileHead = fileHead.toUpperCase();  

        FileType[] fileTypes = FileType.values();  

        for (FileType type : fileTypes) {  
            if (fileHead.startsWith(type.getValue())) {  
                typeNum = isFileType(type);
                return typeNum;
            }  
        }

        //判断上传文件类型是否为0，如果是将其记录下来
        if(typeNum == 0){
            //将内容输出到日志中
            LogUtils.WriteLog("illegalFile", "\n时间"+System.currentTimeMillis()+"、非法文件头："+fileHead+"、文件地址："+path+"\r\n");
        }
        is.close();

        return typeNum;  
    } 
    /** 
     * 得到文件头 
     *  
     * @param is
     *            文件路径 
     * @return 文件头 
     * @throws IOException 
     */  
    public static String getFileContent(InputStream is) throws IOException {  

        byte[] b = new byte[28];  

        InputStream inputStream = null;  

        try {  
            is.read(b, 0, 28);  
        } catch (IOException e) {  
            e.printStackTrace();  
            throw e;  
        } finally {  
            if (inputStream != null) {  
                try {  
                    inputStream.close();  
                } catch (IOException e) {  
                    e.printStackTrace();  
                    throw e;  
                }  
            }  
        }  
        return bytesToHexString(b);  
    }
